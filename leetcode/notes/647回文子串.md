# 647回文子串
```c
int countSubstrings(char * s){
    int l = strlen(s);
    //dp[i][j]存放以s[j]为开始，s[i]为结尾的字符串是否为回文字符串的布尔类型
    bool dp[l][l];
    //存放回文子串的数目
    int S = 0;
    memset(dp, 0, sizeof(dp));
    //这里的遍历循序要注意一下，因为当头尾指针大于1时，要判断dp[i - 1][j + 1]是
    //否为回文串，因此需要确定dp[i - 1][j + 1]是已经判断过的
    //若这里i是逆序遍历，则dp[i - 1][j + 1]是没有被判断的，因此i只能正序遍历
    for(int i = 0; i < l; ++i){
        for(int j = 0; j <= i; ++j){
            //若头字符与尾字符相等，若头尾相差距离小于等于1则为回文子串，
            //若大于1，将头指针前移，尾指针后移，若是回文串，则以s[j]为开始，s[i]为结尾的字符串也为回文串
            if(s[i] == s[j] && (i - j <= 1 || dp[i - 1][j + 1])){
                dp[i][j] = 1;
                ++S;
            }
        }
    }
    return S;
}
````
