# 2269找到一个数字的K美丽值
```cpp
class Solution {
public:
    int divisorSubstrings(int num, int k) {
        string s = to_string(num);   //先转为字符串，找出k长度的字符串，判断是否可以整除
        int n = s.size();
        int res = 0;
        for (int i = 0; i < n - k + 1; ++i) {
            int tmp = stoi(s.substr(i, k));
            if (tmp == 0) continue;
            if (num % tmp == 0) ++res;
        }
        return res;
    }
};
```
